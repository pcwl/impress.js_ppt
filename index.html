#include 

SoftwareSerial mySerial(10, 11);      // RX, TX

void setup()
{
  pinMode(13, OUTPUT);
  digitalWrite(13, LOW);
  // set the data rate for the SoftwareSerial port
  mySerial.begin(115200);
  while (!Serial) {
    ; // wait for serial port to connect. Needed for Leonardo only
  }
  // link server
  mySerial.println("AT+CIPSTART=\"TCP\",\"192.168.2.14\",6001");
}

void loop() // run over and over
{
  char c;
  if(mySerial.available())      // while serial is ready
  {
    c = mySerial.read();      // read serial data char
    Serial.write(c);          // printf the got char

    if(c == 'n'){
      digitalWrite(13, LOW);
    }
    else if(c == 'y'){
      digitalWrite(13, HIGH);
    }
  }

}
